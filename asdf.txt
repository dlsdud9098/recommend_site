function toggleLimitPassRight(targetLink, passSeq) {
	var target = $Element(targetLink);
	if(target.hasClass('down')) { // 닫힌상태
		var oAjax = new $Ajax("/my/cookie/limitPassUsableContentsList.json", {
			method : "GET",
			onload : function(res) {
				var jsonObject = res.json();
				if(jsonObject.success == false){
					alert("일시적인 오류입니다.");
					return;
				}

				var sHtml = $Template("limitPassRightContentListTemplate").process(jsonObject);
				$Element(target).parent().appendHTML(sHtml);

				target.removeClass('down');
				target.addClass('up');
			}
		}).request({
			freePassSeq:passSeq
		});

	} else { // 오픈상태
		target.removeClass('up');
		target.addClass('down');
		target.next().leave();
	}
}

	var pcPossibleYn = bMobileOnly ? 'N' : 'Y';

	if (gLoginStatus == LOGIN_STATUS.LOGOUT) {
		util.login();

function launchViewerWithoutRestore(sProductTypeCode, sUpperOriginalProductId, sOriginalProductId, nProductNo, sRequestLicenseCode, bMobileOnly, idNo) {
	var isPreview = false;
	var pcPossibleYn = bMobileOnly ? 'N' : 'Y';

	if (isValid(sProductTypeCode, bMobileOnly) == false) {
		return;
	}

	launch (sProductTypeCode.toUpperCase(), sUpperOriginalProductId, sOriginalProductId, pcPossibleYn, isPreview, idNo);
}

function launchViewerWithRestore(sProductTypeCode, nProductLevel, sUpperOriginalProductId, sOriginalProductId, nProductNo, nPurchaseNo, sRequestLicenseCode, bMobileOnly, idNo) {
	var isPreview = false;
	var pcPossibleYn = bMobileOnly ? 'N' : 'Y';

	if (isValid(sProductTypeCode, bMobileOnly) == false) {
		return;
	}

	
	var htParam = {'productTypeCode' : sProductTypeCode, 'purchaseNo' : nPurchaseNo, 'productLevel' : nProductLevel, 'originalProductId' : sOriginalProductId, 'upperOriginalProductId' : sUpperOriginalProductId};
	var oAjax = new $Ajax('/my/purchase/bin/restoreBooksProduct.series', {
        method : "POST",
        onload : $Fn(function(req) {
        	var response = req.json();
        	if (!response.success) {
        		console.log("일시적인 오류로 요청하신 작업을 수행할 수 없습니다. \n다시 시도해주세요.");

    		launch (sProductTypeCode, sUpperOriginalProductId, sOriginalProductId, pcPossibleYn, isPreview, idNo);
    		return;
        }, this).bind()
    });
    oAjax.request(htParam);
}

function launchViewerFirstVolumn(sProductTypeCode, sOriginalProductId, sRequestLicenseCode, bMobileOnly) {
	var oAjax = new $Ajax('/comic/firstVolumnHasLicense.series', {
        method : "GET",
        onload : $Fn(function(req) {
        	var response = req.json();
        	if (!response.success) {
        		location.href = "/my/library/productList.series?serviceTypeCode=" + sProductTypeCode;
        		return;
        	}

        	var sVolumnNo = response.firstVolumn;
        	launchViewerWithoutRestore(sProductTypeCode, sOriginalProductId, sVolumnNo, null, sRequestLicenseCode, bMobileOnly);

    		return;
        }, this).bind()
    });
    oAjax.request({'contentId' : sOriginalProductId , 'productTypeCode': sProductTypeCode});
}

function launch (sProductTypeCode, sUpperOriginalProductId, sOriginalProductId, pcPossibleYn, isPreview, idNo) {
	var contentId = sUpperOriginalProductId;
	var volumeNo = sOriginalProductId;

	// 신규 코믹뷰어 실행(MAC OS 환경에서도 실행가능)
	if (sProductTypeCode == 'COMIC') {
		launchNewComicViewer(sProductTypeCode, contentId, volumeNo, pcPossibleYn, isPreview, idNo);
		return;
	}

	// EPub뷰어 실행하기 전 MAC OS 체크
	if($Agent().os().mac){
		alert('구매한 컨텐츠는 Window OS 및 스마트폰/패드류에서 이용 가능합니다. (Mac OS에서는 컨텐츠 이용 불가)');
		return;
	}

	//웹소설
	if (sProductTypeCode == 'NOVEL') {
		launchEpubViewer(contentId, volumeNo, isPreview, pcPossibleYn);
		return;
	}

	//e북
	if (sProductTypeCode == 'EBOOK') {
		contentId = sOriginalProductId;
		launchEpubViewer(contentId, 1, isPreview, pcPossibleYn);
		return;
	}
}

function launchNewComicViewer(productTypeCode, contentsNo, volumeNo, mobileOnly, isPreview, idNo) {

	var isInstalled = false;
	window.focus();
	window.onblur = function () {
		isInstalled = true;
	};

	var newComicViewerLaunchUrl = "series-pc://contents/" + contentsNo + "/volumes/" + volumeNo + "?idNo=" + idNo + "&isPreview=" + isPreview;
	location.assign(newComicViewerLaunchUrl);

	setTimeout(function () {
		if (isInstalled == false) {
			var downloadPage = window.open('/comic/viewer/install/guide');
			if (window.focus) {
				setTimeout(function () {
					downloadPage.focus();
				}, 500);
			}
		}

		window.onblur = null;
	}, 1500);
}
